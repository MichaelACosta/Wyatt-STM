1137  {

1139  //Moves are unusable in this case due to transaction
1140  //restarts. If a movable object is passed into a transaction,
1141  //moved into a WVar and then the transaction gets restarted

1151  at.SetVarValue (m_core_p, std::move (newVal_p));
1152  }
